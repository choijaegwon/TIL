# 폰트 지정 방법

~~~
import UIKit

extension UIFont {
    /// Pretendard 폰트 스타일을 생성합니다
    ///
    /// Pretendard 스타일과 사이즈를 입력받아 Pretendard 폰트 스타일을 생성합니다.
    /// fallback 은 시스템 폰트로 생성하며 사이즈는 동일하게 적용됩니다.
    /// - Parameters:
    ///   - weight: 폰트 굵기, 기본값으로 `regular`
    ///   - size: 폰트 사이즈
    ///
    /// - Returns: 폰트 옵션이 적용된 UIFont
    static func pretendard(fontWeight weight: Pretendard = .regular, fontSize size: CGFloat) -> UIFont {
        if let result = UIFont(name: weight.rawValue, size: size) {
            return result
        } else {
            switch weight {
            case .regular: return .systemFont(ofSize: size, weight: .regular)
            case .medium: return .systemFont(ofSize: size, weight: .medium)
            case .bold: return .systemFont(ofSize: size, weight: .bold)
            }
        }
    }
}
~~~
